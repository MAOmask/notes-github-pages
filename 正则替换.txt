
正则表达式删除空行和注释：
空行：
^\s*\n

注释：
.js & CSS & TS:
(?<!:)\/\/.*|\/\*[\s\S\n]*?\*\/

.html:
<!--[\s\S\n]*?-->

.py:
#.*|"""[\s\S\n]*?"""|'''[\s\S\n]*?'''

.sql:
--.*|/\*[\s\S]*?\*/

Qt.color():
(^|(?<!Qt\.color\())(\s*)(["'])(#[0-9a-fA-F]{3,6})\3
$1$2Qt.color($3$4$3)

\b([\w-]*color[\w-]*\s*:\s*)?(["'])(white|black|red|green|blue|cyan|magenta|yellow|gray|transparent|lightgray|darkgray)\2(?=[^\w]|$)
$1Qt.color($2$3$2)
